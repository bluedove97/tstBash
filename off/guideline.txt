```
# 1. 로컬에서 먼저 테스트
pip install -r requirements.txt
python tools/run_morning_report.py

# 2. Docker 빌드 및 테스트
docker-compose build
docker-compose run --rm app python tools/run_morning_report.py

# 3. 스케줄 확인 (cron 작동 여부)
docker-compose up -d
docker logs content_report_app_1 -f
```
